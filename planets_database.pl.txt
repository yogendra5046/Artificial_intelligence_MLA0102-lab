% b) Saturn’s orbital period (in Earth years) and day length (in Earth hours)

% Add Jupiter to support later distance comparison
planet(jupiter, distance_from_sun(778.5), mass(317.8)).

% --- Rule: Find distance between two planets based on their position from the Sun ---
distance_between(Planet1, Planet2, Distance) :-
    planet(Planet1, distance_from_sun(D1), _);
    planet(Planet1, _, distance_from_sun(D1)),
    planet(Planet2, distance_from_sun(D2), _);
    planet(Planet2, _, distance_from_sun(D2)),
    Distance is abs(D1 - D2).

% Simplified version if distance_from_sun is always defined
distance_between_simple(Planet1, Planet2, Distance) :-
    planet(Planet1, distance_from_sun(D1), _);
    planet(Planet2, distance_from_sun(D2), _),
    Distance is abs(D1 - D2).

% --- d) Find all planets closer to the Sun than a given planet ---
closer_to_sun(Planet, CloserPlanet) :-
    planet(Planet, distance_from_sun(D1), _),
    planet(CloserPlanet, distance_from_sun(D2), _),
    D2 < D1.
